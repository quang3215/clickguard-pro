  import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
  import { getFirestore, collection, addDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js";

    authDomain: "clickfrauddashboard.firebaseapp.com",
    projectId: "clickfrauddashboard",
    storageBucket: "clickfrauddashboard.firebasestorage.app",

  };

  const app = initializeApp(firebaseConfig);
  const db = getFirestore(app);

  async function initClickGuard() {
    try {
      // 1. Lấy địa chỉ IP thực của khách qua API miễn phí
      const ipResponse = await fetch('https://api.ipify.org?format=json');
      const ipData = await ipResponse.json();
      
      // 2. Lấy tên miền hiện tại để phân loại (Multi-tenant)
      const currentDomain = window.location.hostname.replace('www.', '');

      // 3. (Tuỳ chọn) Kiểm tra xem khách có đến từ Google Ads không
      // const urlParams = new URLSearchParams(window.location.search);
      // const isFromAds = urlParams.has('gclid');
      
      // 4. Bắn dữ liệu về máy chủ Firebase của bạn
      await addDoc(collection(db, "visits"), {
        ip: ipData.ip,
        website: currentDomain,
        timestamp: serverTimestamp(),
        userAgent: navigator.userAgent
      });
      
      console.log("ClickGuard: IP Logged successfully");
    } catch (error) {
      console.error("ClickGuard Error:", error);
    }
  }

  // Khởi chạy khi tải trang
  initClickGuard();
</script>
